# Installation:
#   pip install -r requirements-dev.txt
#
# Key tools:
#   - pytest: Run tests with `pytest`
#   - black: Format code with `black .`
#   - flake8: Lint code with `flake8 .`
#   - mypy: Type check with `mypy .`
#   - locust: Load test with `locust`
#   - jupyter: Data analysis with `jupyter notebook`
#
# Pre-commit setup:
#   pre-commit install
#   pre-commit run --all-files
#
# Testing commands:
#   pytest                              # Run all tests
#   pytest --cov=.                      # Run with coverage
#   pytest -x                           # Stop on first failure
#   pytest -v                           # Verbose output
#   pytest tests/test_specific.py       # Run specific test file
#
# Code quality:
#   black . && isort . && flake8 .      # Format and lint
#   mypy . --ignore-missing-imports     # Type checking
#   bandit -r .                         # Security scanning
#   safety check                        # Dependency security
#
# Documentation:
#   sphinx-build -b html docs docs/_build  # Build docs
#   redoc-cli build openapi.json            # API docs
#
# Performance:
#   py-spy top --pid <PID>              # Profile running app
#   memory_profiler script.py           # Memory profiling
#   locust -f locustfile.py             # Load testing
